pragma solidity ^0.8.0;

contract bank{

    uint256 balance = 0;
    address public accOwner;

    constructor() {
        accOwner = msg.sender;
    }


    //function for money deposit
    function deposit() public payable{

        require(accOwner== msg.sender,"YOur are not the owner of this account");
        require(msg.value >0,"you cannnot deposit 0 money ");
        balance += msg.value;
    } 

    //fuuction for widrawal money 
    function widraw() public payable {
         require(accOwner== msg.sender,"YOur are not the owner of this account");
         require(msg.value >0,"you cannnot Widraw 0 unit ");
         balance-=msg.value;


    }
    //function for showing balancce

    function showbalance() public view returns(uint256){

         require(accOwner== msg.sender,"YOur are not the owner of this account");
         return balance;
    }
}
